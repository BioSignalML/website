Signal File

An abstract class defining an interface for accessing signal data files.
object SignalFile {
  val tripleStore = new metadata.jena.TripleStore(config.options)
  }

abstract class SignalFile(uri: URI, create: Boolean) extends metadata.Recording(uri, tripleStore) {
  def open(file: File)
  def open(path: String) = open(new File(path))
  def create(file: File)
  def create(path: String) = create(new File(path))
  def close()
  
  def getSignal(id: String, tm: Interval): SampledSignal
  def putSignal(id: String, data: SampledSegment)
  
  }
